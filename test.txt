class DialogSection(str, Enum):

 TEXT = "TEXT"

 BUTTON = "BUTTON"



class ReferenceType(str, Enum):

 URL = "URL"

 DIALOG = "DIALOG"



@dataclass

class ValidationError:

 dialog_name: str

 answer_index: Optional[int]

 section: DialogSection

 section_index: Optional[int]

 reference_type: ReferenceType

 value: str

 error_message: str



class Report:

 def __init__(self, logger: Optional[logging.Logger] = None):

   self.errors: list[ValidationError] = []
   self.logger = logger



 def write_to_csv(self, output_path="report.csv"):

   if not self.errors:

     if self.logger:

       self.logger.info("No errors found - CSV will not be generated")

     return



   with open(output_path, "w", newline="", encoding="utf-8") as f:

     writer = csv.writer(f)


writer.writerow(["dialog_name","answer_index","section","section_index","reference_type"
,"value","error"])

     for error in self.errors:

       writer.writerow([

            error.dialog_name,

            error.answer_index if error.answer_index is not None else None,

            error.section.value,

            error.section_index if error.section_index is not None else None,

            error.reference_type.value,

            error.value,

            error.error_message

       ])



   if self.logger:

     self.logger.info(f"CSV report generated: {output_path}")



 def write_to_json(self, output_path="report.json"):

   if not self.errors:
     if self.logger:

       self.logger.info("No errors found - JSON will not be generated")

     return



   with open(output_path, "w", encoding="utf-8") as f:

     json.dump([{

       "dialog_name": e.dialog_name,

       "answer_index": e.answer_index,

       "section": e.section.value,

       "section_index": e.section_index,

       "reference_type": e.reference_type.value,

       "value": e.value,

       "error_message": e.error_message

     } for e in self.errors], f, ensure_ascii=False, indent=4)



   if self.logger:

     self.logger.info(f"JSON report generated: {output_path}")

# Utility functions

def _is_url(value: str) -> bool:

  return value.startswith("http://") or value.startswith("https://")



def _extract_urls_from_text(text: str) -> List[str]:

  url_regex = r'\[.*?\]\((https?://[^)]+)\)'

  return re.findall(url_regex, text)



def _dialog_exists(dialog_name: str, dialogs_dir: str) -> bool:

  return any(

      os.path.isfile(os.path.join(dialogs_dir, f"{dialog_name}{ext}"))

      for ext in (".yml", ".yaml")

  )



@lru_cache(maxsize=64)

def _check_url(url: str, timeout=5) -> bool:

  try:

      response = requests.head(url, allow_redirects=True, timeout=timeout)
   if response.status_code >= 400:

     response = requests.get(url, allow_redirects=True, timeout=timeout)



   return response.status_code < 400

 except requests.RequestException:

   logger.warning(f"URL check failed for: {url}")

   return False



def validate_answer_text(dialog_name: str, answer_idx: int, text: str, report: Report):

 urls_in_text = _extract_urls_from_text(text)

 for url_idx, url in enumerate(urls_in_text):

   if not _check_url(url):

     report.add(

         dialog_name=dialog_name,

         answer_index=answer_idx,

         section=DialogSection.TEXT,

         section_index=url_idx,

         reference_type=ReferenceType.URL,

         value=url,

         error_message="URL is not reachable"

     )



def validate_answer_button(

 dialog_name: str,

 answer_idx: int,

 button: Button,

 button_idx: int,

 dialogs_dir: str,
     report: Report,

):

     payload = button.payload



     if _is_url(payload):

       if not _check_url(payload):

         report.add(

             dialog_name=dialog_name,

             answer_index=answer_idx,

             section=DialogSection.BUTTON,

             section_index=button_idx,

             reference_type=ReferenceType.URL,

             value=payload,

             error_message="URL is not reachable",

         )

     else:

       if not _dialog_exists(payload, dialogs_dir):

         report.add(

             dialog_name=dialog_name,

             answer_index=answer_idx,

             section=DialogSection.BUTTON,

             section_index=button_idx,

             reference_type=ReferenceType.DIALOG,

             value=payload,

             error_message="Referenced dialog does not exist",

         )

def fetch_file(filepath: str, branch: str = "master") -> requests.Response:

  """

  Take a relative filepath in the repository and fetch the file over https

  """

  URL = f"https://raw.code.rbi.tech/raiffeisen/maia-bot-rbhr-raia/{branch}/{filepath}"

  token = os.getenv('GITHUB_PAT')



  if not token:

   raise RuntimeError("GitHub Personal Access Token (GITHUB_PAT) is not set in the
environment variables.")



  headers = {

      "Authorization": f"token {token}"

  }



  try:

      response = requests.get(URL, headers=headers)

      response.raise_for_status()

      return response

  except requests.exceptions.RequestException as e:

      raise RuntimeError(f"Failed to fetch file from {URL}") from e

def get_files_and_dirs(dir: str, branch: str = "main") -> dict[str, list[str]]:

  """

  Retrieve a list of files and directories from a specified directory in a GitHub repository.

  """

  URL = f"https://api.code.rbi.tech/repos/raiffeisen/maia-bot-rbhr-
raia/contents/{dir}?ref={branch}"



  token = os.getenv('GITHUB_PAT')



  if not token:

   raise RuntimeError("GitHub Personal Access Token (GITHUB_PAT) is not set in the
environment variables.")



  headers = {

      "Authorization": f"token {token}"

  }



  try:

      categorized_data = {'files': [], 'dirs': []}

      response = requests.get(URL, headers=headers)

      response.raise_for_status()

      for item in response.json():

         if item['type'] == 'file':

           categorized_data['files'].append(item['path'])
        elif item['type'] == 'dir':

          categorized_data['dirs'].append(item['path'])

   return categorized_data

 except requests.exceptions.RequestException as e:

   raise RuntimeError(f"Failed to fetch file from {URL}") from e

import subprocess



def pull_repository(repo_url: str, branch: str = "main", bin_dir: str = "bin") -> str:

  """

  Clone the repository and checkout the specified branch into the bin directory.

  Each branch is stored in its own folder.

  """

  repo_name = repo_url.split('/')[-1]

  branch_dir = os.path.join(bin_dir, repo_name, branch)



  if not os.path.exists(branch_dir):

    os.makedirs(branch_dir)
  try:

    subprocess.run(

         ["git", "clone", "--branch", branch, repo_url, branch_dir],

         check=True

    )

  except subprocess.CalledProcessError as e:

    raise RuntimeError(f"Failed to clone repository: {e}")



  return branch_dir



def refresh(repo_url: str, branch: str = "master", bin_dir: str = "bin") -> str:

  """

  Check for updates in the specified branch and pull if there are changes.

  """

  repo_name = repo_url.split('/')[-1]

  branch_dir = os.path.join(bin_dir, repo_name, branch)



  if not os.path.exists(branch_dir):

    # If the directory doesn't exist, clone the repository

    return pull_repository(repo_url, branch, bin_dir)



  try:

    subprocess.run(

         ["git", "-C", branch_dir, "pull", "origin", branch],

         check=True

    )

  except subprocess.CalledProcessError as e:
     raise RuntimeError(f"Failed to pull changes for repository: {e}")



 return branch_dir

class SimpleColorFormatter(logging.Formatter):

 LEVEL_COLORS = {

     logging.DEBUG: "\033[37m",

     logging.INFO: "\033[32m",

     logging.WARNING: "\033[33m",

     logging.ERROR: "\033[31m",

     logging.CRITICAL: "\033[1;31m"

 }



 RESET = "\033[0m"



 def format(self, record):

     color = self.LEVEL_COLORS.get(record.levelno, "")

     message = super().format(record)

     return f"{color}{message}{self.RESET}"
def use_logging(name: str, log_file: str = None) -> logging.Logger:

 logger = logging.getLogger(name)

 logger.setLevel(logging.DEBUG)



 console_formatter = SimpleColorFormatter(

     fmt="%(asctime)s [%(levelname)s] %(name)s - %(message)s",

     datefmt="%Y-%m-%d %H:%M:%S"

 )



 console_handler = logging.StreamHandler()

 console_handler.setLevel(logging.INFO)

 console_handler.setFormatter(console_formatter)

 logger.addHandler(console_handler)



 if log_file:

     os.makedirs(os.path.dirname(log_file), exist_ok=True)

     file_formatter = logging.Formatter(

         fmt="%(asctime)s [%(levelname)s] %(name)s - %(message)s",

         datefmt="%Y-%m-%d %H:%M:%S"

     )

     file_handler = logging.FileHandler(log_file, encoding="utf-8")

     file_handler.setLevel(logging.DEBUG)

     file_handler.setFormatter(file_formatter)

     logger.addHandler(file_handler)



 return logger

def extract_urls_from_text(text: str) -> List[str]:

  if not text:

    return []
  md_urls = re.findall(r'\\[.*?\\]\\((https?://[^\s)]+)\\)', text)

  plain_urls = re.findall(r'https?://[^\s)]+', text)



  return list(set(md_urls + plain_urls))



def collect_urls(dialog: Dialog) -> List[Tuple[str, str, str]]:

  urls_info = []

  for answer in dialog.answers:



    for url in extract_urls_from_text(answer.text):

      urls_info.append((dialog.name, url, "TEXT"))



    for button in answer.buttons:

      if button.payload and _is_url(button.payload):

        urls_info.append((dialog.name, button.payload, "BUTTON"))

  return urls_info
